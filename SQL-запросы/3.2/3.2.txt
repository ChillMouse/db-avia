/*3.2. Сформировать запросы, которые бы позволили пользователю сориентироваться в перечне туров (использовать фильтры: курорт/отель, питание, цена).*/

SELECT * /* Выбираем все столбцы для вывода */
FROM /* Работаем с этой таблицей, таблица составлена "на лету" из четырёх по соответсвию внешних и первичных ключей */
(
(
Tours A /*Таблица с турами с псевдонимом "А"*/
INNER JOIN /*Присоединение с учётом соответсвия полей */
TypeEatHotel Eat ON (Eat.id = A.id_typeEat)) /* Таблицу TypeEatHotel  присоединяем к Tours по соответствию полей "Eat.id" и "A.id_typeEat" */
INNER JOIN /*Присоединение с учётом соответсвия полей */
Hotels ON (A.id_hotel = Hotels.id) /*Таблицу Hotels присоединяем к предыдущим по соответствию полей "Tours.id_hotel" и "Hotels.id" */
INNER JOIN /*Присоединение с учётом соответсвия полей */
Prices ON (Prices.id_tour = A.id) /*Таблицу Prices присоединяем к предыдущим по соответствию полей "Prices.id_tour" и "Tours.id" */
) 
WHERE /*Здесь описаны условия, они являются фильтрами*/
Hotels.city LIKE '%' AND Hotels.title LIKE '%Sukko%' /* Фильтр по городу-курорту и названию отеля*/
AND /*Логическое И*/
Eat.id = 1 /* Фильтр по питанию. Питание находится либо в отеле, либо где-то за границей территории отеля. Предполагается, что на сайте будет выпадающий список с выбором, где ID будет подставляться сам*/
AND /*Логическое И*/
Prices.priceTotal > 1000 AND Prices.priceTotal < 1000000 /* Фильтр по цене. Формат: ОТ и ДО. По минимальной и по максимальной цене*/

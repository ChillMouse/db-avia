/* 3.8. Создать запрос, который будет отображать "Рекомендации туроператора», то есть те туры, стоимость которых меньше средней */


SELECT * /* Выбираем все столбцы */
FROM Tours INNER JOIN Prices ON (Tours.id = Prices.id_tour) /* выбираем столбцы из слияния двух таблиц: Tours и Prices, в Tours - информация о турах, в Prices - цены на тур.*/
WHERE priceTotal < ( /* Условие: если цена на тур меньше, чем средняя цена на тур*/
	SELECT AVG(priceTotal) /* Получение константы - средняя цена на тур */
	FROM Prices /* получение данных из таблицы Prices*/
)
